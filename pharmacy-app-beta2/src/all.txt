import { useState } from "react"
import axios from "axios"
import PharmacyMap from "./components/PharmacyMap"

// ✅ FIXED PORT
const API = "http://localhost:3000"

// ✅ TYPE (avoid bugs)
type Pharmacy = {
  pharmacy: string
  city: string
  latitude: number
  longitude: number
  price: number
  stock: number
  amo_supported: boolean
}

function App() {
  const [query, setQuery] = useState("")
  const [suggestions, setSuggestions] = useState<string[]>([])
  const [results, setResults] = useState<Pharmacy[]>([])

  async function fetchSuggestions(q: string) {
    if (q.length < 2) {
      setSuggestions([])
      return
    }

    try {
      const res = await axios.get(`${API}/api/suggestions?q=${q}`)
      setSuggestions(res.data)
    } catch (err) {
      console.error("Suggestion error:", err)
    }
  }

  async function searchDrug(name: string) {
    try {
      const res = await axios.get(`${API}/api/aggregate-stock/${name}`)
      setResults(res.data)
    } catch (err) {
      console.error("Search error:", err)
    }
  }

  return (
    <div style={{ padding: 40, fontFamily: "Arial" }}>
    <h1>💊 Pharmacy Finder</h1>

    {/* SEARCH INPUT */}
    <input
    style={{ padding: 10, width: 300, fontSize: 16 }}
    placeholder="Search medicine..."
    value={query}
    onChange={(e) => {
      setQuery(e.target.value)
      fetchSuggestions(e.target.value)
    }}
    />

    {/* SUGGESTIONS */}
    <div>
    {suggestions.map((s, i) => (
      <div
      key={i}
      style={{
        cursor: "pointer",
        padding: 5,
        borderBottom: "1px solid #ddd",
        background: "#f9f9f9"
      }}
      onClick={() => {
        setQuery(s)
        setSuggestions([])
        searchDrug(s)
      }}
      >
      {s}
      </div>
    ))}
    </div>

    <hr style={{ margin: "30px 0" }} />

    {/* RESULTS */}
    <h2>Available Pharmacies</h2>

    {results.length === 0 && <p>No results found</p>}

    {results.map((r, i) => (
      <div
      key={i}
      style={{
        border: "1px solid #ccc",
        padding: 15,
        marginBottom: 10,
        borderRadius: 8
      }}
      >
      <h3>{r.pharmacy}</h3>
      <p>City: {r.city}</p>
      <p>Price: {r.price} FCFA</p>
      <p>Stock: {r.stock}</p>
      <p>AMO: {r.amo_supported ? "✅ Supported" : "❌ Not Supported"}</p>
      </div>
    ))}

    {/* 🗺️ MAP */}
    {results.length > 0 && <PharmacyMap pharmacies={results} />}
    </div>
  )
}

export default App

import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";
import "leaflet/dist/leaflet.css";
import "../utils/fixLeafletIcon";

type Pharmacy = {
  pharmacy: string;
  latitude: number;
  longitude: number;
  price: number;
  stock: number;
  amo_supported?: boolean;
};

export default function PharmacyMap({ pharmacies }: { pharmacies: Pharmacy[] }) {
  if (pharmacies.length === 0) return null;

  const center: [number, number] = [
    pharmacies[0].latitude,
    pharmacies[0].longitude,
  ];

  return (
    <MapContainer
      center={center}
      zoom={13}
      style={{ height: "400px", width: "100%", marginTop: "20px" }}
    >
      <TileLayer
        attribution="&copy; OpenStreetMap contributors"
        url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
      />

      {pharmacies.map((p, index) => (
        <Marker key={index} position={[p.latitude, p.longitude]}>
          <Popup>
            <strong>{p.pharmacy}</strong> <br />
            💰 {p.price} FCFA <br />
            📦 Stock: {p.stock} <br />
            {p.amo_supported ? "✅ AMO" : "❌ No AMO"}
          </Popup>
        </Marker>
      ))}
    </MapContainer>
  );
}
